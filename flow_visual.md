### –ö–æ–º–ø–æ–Ω–µ–Ω—Ç 1: –ó–≤—É–∫–æ–≤–æ–π –õ–∞–Ω–¥—à–∞—Ñ—Ç (–ê—É–¥–∏–æ)

### –ö–æ–º–ø–æ–Ω–µ–Ω—Ç 2: –í–∏–∑—É–∞–ª—å–Ω–∞—è –°—Ä–µ–¥–∞ (–ê–Ω–∏–º–∞—Ü–∏—è "Reflect")

* **–ö–æ–¥–æ–≤–æ–µ –ò–º—è:** "Neural Plasma" (–ù–µ–π—Ä–æ–Ω–Ω–∞—è –ü–ª–∞–∑–º–∞)
* **–¢–µ—Ö–Ω–∏—á–µ—Å–∫–∏–π –ü–æ–¥—Ö–æ–¥ (–®–µ–π–¥–µ—Ä GLSL/WGSL):**
    –ú—ã –Ω–µ –∏—Å–ø–æ–ª—å–∑—É–µ–º –ø—Ä–æ—Å—Ç–æ —à—É–º –∏–ª–∏ –ø—Ä–æ—Å—Ç–æ –º–µ—Ç–∞–±–æ–ª—ã. –ú—ã –∏—Å–ø–æ–ª—å–∑—É–µ–º **–≥–∏–±—Ä–∏–¥–Ω—ã–π –ø–æ–¥—Ö–æ–¥**:
    * **–ú–µ—Ç–∞–±–æ–ª—ã (–Ø–¥—Ä–∞):** 3-5 "–≥–æ—Ä—è—á–∏—Ö —Ç–æ—á–µ–∫" (`vec2 core_points`), –∫–æ—Ç–æ—Ä—ã–µ –º–µ–¥–ª–µ–Ω–Ω–æ –¥—Ä–µ–π—Ñ—É—é—Ç –ø–æ —ç–∫—Ä–∞–Ω—É. –≠—Ç–æ —Å—Ç—Ä—É–∫—Ç—É—Ä–∞.
    * **–ü–æ–ª–µ –¢–µ—á–µ–Ω–∏—è (–ò—Å–∫–∞–∂–µ–Ω–∏–µ):** –ú–Ω–æ–≥–æ—Å–ª–æ–π–Ω—ã–π –ø—Ä–æ—Ü–µ–¥—É—Ä–Ω—ã–π —à—É–º (FBM) –∏—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è –∫–∞–∫ "–ø–æ–ª–µ —Ç–µ—á–µ–Ω–∏—è" (Flow Field), –∫–æ—Ç–æ—Ä–æ–µ *–∏—Å–∫–∞–∂–∞–µ—Ç* –ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–æ (`uv`), –≤ –∫–æ—Ç–æ—Ä–æ–º "–∂–∏–≤—É—Ç" –º–µ—Ç–∞–±–æ–ª—ã.
    * **–¢–µ–∫—Å—Ç—É—Ä–∞ (–û–∫—Ä–∞—à–∏–≤–∞–Ω–∏–µ):** –í—Ç–æ—Ä–æ–π, –±–æ–ª–µ–µ –º–µ–ª–∫–æ–∑–µ—Ä–Ω–∏—Å—Ç—ã–π —Å–ª–æ–π FBM –∏—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è –¥–ª—è —Å–æ–∑–¥–∞–Ω–∏—è —ç—Ñ—Ñ–µ–∫—Ç–∞ "–ø–µ—Ä–µ–ª–∏–≤–∞—é—â–∏—Ö—Å—è —á–µ—Ä–Ω–∏–ª" –≤–Ω—É—Ç—Ä–∏ –ø–ª–∞–∑–º—ã.

#### –ö–ª—é—á–µ–≤–æ–π –§—Ä–∞–≥–º–µ–Ω—Ç –®–µ–π–¥–µ—Ä–∞ (GLSL –ö–æ–Ω—Ü–µ–ø—Ç):

```glsl
// ... —Ñ—É–Ω–∫—Ü–∏–∏ snoise() –∏ fbm() ...

    vec2 uv = gl_FragCoord.xy / u_resolution.xy;
    
    // 1. –ü–û–õ–ï –¢–ï–ß–ï–ù–ò–Ø (–ò—Å–∫–∞–∂–µ–Ω–∏–µ)
    // –ò—Å–ø–æ–ª—å–∑—É–µ–º 4D-—à—É–º (x, y, time) –¥–ª—è —Å–æ–∑–¥–∞–Ω–∏—è "–≤–µ—Ç—Ä–∞"
    vec4 noise_vec = vec4(uv * 2.0, time * 0.1, 0.0);
    float noise = fbm(noise_vec);
    
    // –ò—Å–∫–∞–∂–∞–µ–º UV-–∫–æ–æ—Ä–¥–∏–Ω–∞—Ç—ã —ç—Ç–∏–º —à—É–º–æ–º
    vec2 distorted_uv = uv + vec2(noise * 0.1, noise * 0.1);

    // 2. –Ø–î–†–ê (–ú–µ—Ç–∞–±–æ–ª—ã)
    // –°—á–∏—Ç–∞–µ–º —ç–Ω–µ—Ä–≥–∏—é –≤ *–∏—Å–∫–∞–∂–µ–Ω–Ω–æ–º* –ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–µ
    float energy = 0.0;
    float radius = 0.15;
    for (int i = 0; i < 4; i++) {
        float dist = distance(distorted_uv, core_points[i]);
        energy += smoothstep(radius, 0.0, dist); // –§–æ—Ä–º—É–ª–∞ –º–µ—Ç–∞–±–æ–ª–∞
    }

    // 3. –û–ö–†–ê–®–ò–í–ê–ù–ò–ï (–¢–µ–∫—Å—Ç—É—Ä–∞)

    if (plasma_shape == 0.0) {
        // –§–æ–Ω: –≥–ª—É–±–æ–∫–∏–π, —Ç–µ–º–Ω—ã–π, –≤ —Å—Ç–∏–ª–µ "dark gothic opium"
        gl_FragColor = vec4(0.01, 0.0, 0.02, 1.0); // –ë–∞–≥—Ä–æ–≤–æ-—á–µ—Ä–Ω—ã–π
        return;
    }

    // –í–Ω—É—Ç—Ä–µ–Ω–Ω—è—è —Ç–µ–∫—Å—Ç—É—Ä–∞ –ø–ª–∞–∑–º—ã (–≤—Ç–æ—Ä–æ–π, –º–µ–¥–ª–µ–Ω–Ω—ã–π —à—É–º)

    float texture_noise = fbm(texture_noise_vec);

    // –°–º–µ—à–∏–≤–∞–µ–º –¥–≤–∞ "—á–µ—Ä–Ω–∏–ª—å–Ω—ã—Ö" —Ü–≤–µ—Ç–∞
    vec3 colorA = vec3(0.1, 0.0, 0.3); // –ì–ª—É–±–æ–∫–∏–π —Ñ–∏–æ–ª–µ—Ç–æ–≤—ã–π
    vec3 colorB = vec3(0.8, 0.1, 0.2); // –¢–µ–º–Ω–æ-–∫—Ä–∞—Å–Ω—ã–π
    vec3 ink_color = mix(colorA, colorB, smoothstep(-0.5, 0.5, texture_noise));

    float brightness = smoothstep(0.4, 1.0, energy); // –í–Ω—É—Ç—Ä–µ–Ω–Ω–µ–µ —Å–≤–µ—á–µ–Ω–∏–µ
    
    vec3 final_color = ink_color * (brightness * 2.0);
    gl_FragColor = vec4(final_color, 1.0);
}
